dpma_general_settings.auth_option = Opciones de Autenticación
dpma_general_settings.config_auth = Configuración de la Autenticación
dpma_general_settings.config_auth.disabled = Deshabilitado
dpma_general_settings.config_auth.globalpin = PIN Global
dpma_general_settings.config_auth.mac = Mac
dpma_general_settings.config_auth.pin = Pin
dpma_general_settings.config_auth.tooltip = Especifica el tipo de autenticación que se requiere para un teléfono para extraer una configuración válida desde el servidor para un usuario. A continuación se presentan los valores válidos para esta opción.
dpma_general_settings.config_auth.tooltip.disabled = Configuraciones se sirven sin autenticación. Cualquier teléfono puede extraer cualquier configuración de teléfono definido en este archivo sin problemas de autenticación.
dpma_general_settings.config_auth.tooltip.globalpin = Sólo la contraseña global (o group_pin, si se establece en un teléfono) deben ser válidos para recuperar la configuración.
dpma_general_settings.config_auth.tooltip.mac = Solo la dirección MAC debe coincidir con las peticiones de configuración del teléfono.
dpma_general_settings.config_auth.tooltip.pin = Sólo contraseña específica del teléfono debe ser válido para recuperar la configuración solicitada.
dpma_general_settings.file_directory = Directorio de Archivos
dpma_general_settings.file_directory.tooltip = Los directorio de archivos apuntan a las carpetas de los teléfonos permitiendo recuperar archivos TXT y binarios. De forma predeterminada, este directorio se encuentra en los teléfonos /var/lib/asterisk/digium_phones.
dpma_general_settings.general = General
dpma_general_settings.globalpin = PIN Global
dpma_general_settings.globalpin.tooltip = Si el userlist_auth se establece a globalpin, define el PIN que debe introducirse en un teléfono, en el arranque, con el fin de recuperar la lista de configuraciones de teléfono disponibles. Si config_auth se establece en globalpin, define el PIN que debe introducirse en un teléfono, en la pantalla de lista de usuarios, con el fin de solicitar una configuración de teléfono en particular - notar que si ya se ha introducido el globalpin para autenticar y recuperar la lista de configuraciones de teléfonos disponibles, que no se requiere una segunda vez con el fin de solicitar una configuración de teléfono en particular. Si se utiliza group_pin para los teléfonos, y userlist_auth y config_auth se establece en globalpin, a continuación, la entrada de la globalpin mostrará todos los teléfonos para todos los grupos.
dpma_general_settings.mdns = Opciones de Descubrimiento de Servicios mDNS
dpma_general_settings.mdns_address = Dirección de Registro
dpma_general_settings.mdns_address.tooltip = La dirección de registro proporcionada por el DMPA a Avahi. Esta es la dirección a la que los teléfonos que descubren los que utilizan mDNS DPMA se pondrán en contacto.
dpma_general_settings.mdns_port = Puerto de Registro
dpma_general_settings.mdns_port.tooltip = El puerto de registro proporcionado por el DMPA a Avahi. Este es el puerto al que los teléfonos que descubren los que utilizan mDNS DPMA se conectarán.
dpma_general_settings.other_options = Otras Opciones
dpma_general_settings.pjsip_message_context = Mensaje de Contexto PJSIP
dpma_general_settings.pjsip_message_context.tooltip = El contexto utilizado para el encaminamiento de mensajes en PJSIP para que los conflictos que se encuentran en pjsip.conf pueden ser evitados. El valor predeterminado dpma_pjsip_message_context.
dpma_general_settings.server_uuid = Identificador de Servidor
dpma_general_settings.server_uuid.tooltip = Establece un identificador único para este servidor. Se utiliza, sobre todo, dentro de la aplicación de correo de voz para su transmisión a los contactos asociados con este servidor. Opcional, no tiene valor predeterminado. Si server_uuid se establece en la parte general, sino que también debe estar configurado para contactos individuales. Si sólo se establece en un lugar (general o de contacto) en lugar de los dos, no se permite el envío de mensajes de voz a los contactos.
dpma_general_settings.service_discovery_enabled = Descubrimiento de Servicio
dpma_general_settings.service_discovery_enabled.tooltip = Si desea o no hacer publicidad utilizando el servidor de configuración Avahi. Por defecto esta opción se establece en sí. El servicio de descubrimiento sólo se habilita cuando el servidor de registro y el puerto están configurados de forma explícita.
dpma_general_settings.service_name = Nombre del Servicio
dpma_general_settings.service_name.placeholder = DPMA de Configuración del Servidor
dpma_general_settings.service_name.tooltip = El nombre del servicio que se utilizará en el descubrimiento de servicios de mDNS. Esto será por defecto "Teléfonos Digium configuración del servidor" cuando el Descubrimiento de Servicio se establece y la dirección de registro y el puerto están presentes.
dpma_general_settings.userlist_auth = Autenticación Lista de Usuarios
dpma_general_settings.userlist_auth.disabled = Deshabilitado
dpma_general_settings.userlist_auth.globalpin = PIN Global
dpma_general_settings.userlist_auth.tooltip = Especifica el tipo de autenticación que se requiere para un teléfono para extraer de la lista de usuarios. Este valor está en "Global Pin" requiere el usuario del teléfono para introducir el valor globalpin antes de ser autorizados para obtener la lista de usuarios.
